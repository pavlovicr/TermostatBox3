37/* GIF Defines and variables */
38#define MAX_CHUNK_SIZE 255
39//
40// These 2 macros can be changed to limit the amount of RAM
41// required by the decoder. For example, decoding 1-bit images to
42// a 128x32 display will not need a max code size of 12 nor a palette
43// with 256 entries
44//
45#define TURBO_BUFFER_SIZE 0x6100
46
47// If you intend to decode generic GIFs, you want this value to be 12. If you are using GIFs solely for animations in
48// your own project, and you control the GIFs you intend to play, then you can save additional RAM here:
49// the decoder must reserve a minimum of 4 byte * (1<<MAX_CODE_SIZE) for the dictionary, but based on implementation
50// actually reserves 5 byte * (1<<MAX_CODE_SIZE). Small or low colour GIFs may inherently not require a large
51// dictionary. For larger GIFs, the en(!)coder can "voluntarily" choose not to utilize the entire dictionary. I.e.,
52// by preparing (specially encoding) the GIFs, you can save >10kB RAM, but you will not be able to decode arbitrary
53// images anymore. One application to craft such GIFs can be found here (use option -d)
54// https://create.stephan-brumme.com/flexigif-lossless-gif-lzw-optimization/
55#define MAX_CODE_SIZE 12
56
57#define MAX_COLORS 256
58#define MAX_WIDTH 480
59#define LZW_BUF_SIZE (6*MAX_CHUNK_SIZE)
60#define LZW_HIGHWATER (4*MAX_CHUNK_SIZE)
61// This buffer is used to store the pixel sequence in reverse order
62// it needs to be large enough to hold the longest possible
63// sequence (1<<MAX_CODE_SIZE)
64#define FILE_BUF_SIZE (1<<MAX_CODE_SIZE)

68#define LINK_UNUSED 5911 // 0x1717 to use memset
69#define LINK_END 5912
70#define MAX_HASH 5003
71// expanded LZW buffer for Turbo mode
72#define LZW_BUF_SIZE_TURBO (LZW_BUF_SIZE + (2<<MAX_CODE_SIZE) + (PIXEL_LAST*2) + MAX_WIDTH)
73#define LZW_HIGHWATER_TURBO ((LZW_BUF_SIZE_TURBO * 14) / 16)

78enum {
79 GIF_PALETTE_RGB565_LE = 0, // little endian (default)
80 GIF_PALETTE_RGB565_BE, // big endian
81 GIF_PALETTE_RGB888, // original 24-bpp entries
82 GIF_PALETTE_RGB8888, // 32-bit (alpha = 0xff or 0x00)
83 GIF_PALETTE_1BPP, // 1-bit per pixel (horizontal, MSB on left)
84 GIF_PALETTE_1BPP_OLED // 1-bit per pixel (vertical, LSB on top)
85};
86// for compatibility with older code
87#define LITTLE_ENDIAN_PIXELS GIF_PALETTE_RGB565_LE
88#define BIG_ENDIAN_PIXELS GIF_PALETTE_RGB565_BE
89//
90// Draw types
91//
92// RAW = 8-bit palettized pixels requiring transparent pixel handling and conversion through the palette.
93// Each line is sent to the GIFDraw callback as 8-bit pixels. If a framebuffer exists, the lines will be
94// written there too. The GIFDraw callback is optional if there is a framebuffer allocated.
95//
96// COOKED = 16/24/32-bpp fully rendered pixels ready for display. This requires a full frame buffer with extra
97// room for the fully rendered pixels at the end of the 8-bit pixel buffer. For example, a 160x120
98// canvas size with 24-bit output would require (160*120 + 3*160) bytes.
99// Each prepared line is sent to the GIFDraw callback as a row of 16/24/32-bit pixels.
100//
101enum {
102 GIF_DRAW_RAW = 0,
103 GIF_DRAW_COOKED
104};
