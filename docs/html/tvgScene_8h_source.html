34struct SceneIterator : Iterator
35{
36 list<Paint*>* paints;
37 list<Paint*>::iterator itr;
38
39 SceneIterator(list<Paint*>* p) : paints(p)
40 {
41 begin();
42 }
43
44 const Paint* next() override
45 {
46 if (itr == paints->end()) return nullptr;
47 auto paint = *itr;
48 ++itr;
49 return paint;
50 }
51
52 uint32_t count() override
53 {
54 return paints->size();
55 }
56
57 void begin() override
58 {
59 itr = paints->begin();
60 }
61};
62
63struct Scene::Impl
64{
65 list<Paint*> paints;
66 RenderData rd = nullptr;
67 Scene* scene = nullptr;
68 RenderRegion vport = {0, 0, INT32_MAX, INT32_MAX};
69 Array<RenderEffect*>* effects = nullptr;
70 uint8_t opacity; //for composition
71 bool needComp = false; //composite or not
72
73 Impl(Scene* s) : scene(s)
74 {
75 }
76
77 ~Impl()
78 {
79 resetEffects();
80
81 for (auto paint : paints) {
82 if (P(paint)->unref() == 0) delete(paint);
83 }
84
85 if (auto renderer = PP(scene)->renderer) {
86 renderer->dispose(rd);
87 }
88 }
89
90 bool needComposition(uint8_t opacity)
91 {
92 if (opacity == 0 || paints.empty()) return false;
93
94 //post effects requires composition
95 if (effects) return true;
96
97 //Masking may require composition (even if opacity == 255)
98 auto compMethod = scene->composite(nullptr);
99 if (compMethod != CompositeMethod::None && compMethod != CompositeMethod::ClipPath) return true;
100
101 //Blending may require composition (even if opacity == 255)
102 if (PP(scene)->blendMethod != BlendMethod::Normal) return true;
103
104 //Half translucent requires intermediate composition.
105 if (opacity == 255) return false;
106
107 //If scene has several children or only scene, it may require composition.
108 //OPTIMIZE: the bitmap type of the picture would not need the composition.
109 //OPTIMIZE: a single paint of a scene would not need the composition.
110 if (paints.size() == 1 && paints.front()->type() == Type::Shape) return false;
111
112 return true;
113 }
114
115 RenderData update(RenderMethod* renderer, const Matrix& transform, Array<RenderData>& clips, uint8_t opacity, RenderUpdateFlag flag, TVG_UNUSED bool clipper)
116 {
117 this->vport = renderer->viewport();
118
119 if ((needComp = needComposition(opacity))) {
120 /* Overriding opacity value. If this scene is half-translucent,
121 It must do intermediate composition with that opacity value. */
122 this->opacity = opacity;
123 opacity = 255;
124 }
125 for (auto paint : paints) {
126 paint->pImpl->update(renderer, transform, clips, opacity, flag, false);
127 }
128
129 return nullptr;
130 }
131
132 bool render(RenderMethod* renderer)
133 {
134 RenderCompositor* cmp = nullptr;
135 auto ret = true;
136
137 renderer->blend(PP(scene)->blendMethod);
138
139 if (needComp) {
140 cmp = renderer->target(bounds(renderer), renderer->colorSpace());
141 renderer->beginComposite(cmp, CompositeMethod::None, opacity);
142 }
143
144 for (auto paint : paints) {
145 ret &= paint->pImpl->render(renderer);
146 }
147
148 if (cmp) {
149 //Apply post effects if any.
150 if (effects) {
151 for (auto e = effects->begin(); e < effects->end(); ++e) {
152 renderer->effect(cmp, *e);
153 }
154 }
155 renderer->endComposite(cmp);
156 }
157
158 return ret;
159 }
160
161 RenderRegion bounds(RenderMethod* renderer) const
162 {
163 if (paints.empty()) return {0, 0, 0, 0};
164
165 int32_t x1 = INT32_MAX;
166 int32_t y1 = INT32_MAX;
167 int32_t x2 = 0;
168 int32_t y2 = 0;
169
170 for (auto paint : paints) {
171 auto region = paint->pImpl->bounds(renderer);
172
173 //Merge regions
174 if (region.x < x1) x1 = region.x;
175 if (x2 < region.x + region.w) x2 = (region.x + region.w);
176 if (region.y < y1) y1 = region.y;
177 if (y2 < region.y + region.h) y2 = (region.y + region.h);
178 }
179
180 //Extends the render region if post effects require
181 int32_t ex = 0, ey = 0, ew = 0, eh = 0;
182 if (effects) {
183 for (auto e = effects->begin(); e < effects->end(); ++e) {
184 auto effect = *e;
185 if (effect->rd || renderer->prepare(effect)) {
186 ex = std::min(ex, effect->extend.x);
187 ey = std::min(ey, effect->extend.y);
188 ew = std::max(ew, effect->extend.w);
189 eh = std::max(eh, effect->extend.h);
190 }
191 }
192 }
193
194 auto ret = RenderRegion{x1 + ex, y1 + ey, (x2 - x1) + ew, (y2 - y1) + eh};
195 ret.intersect(this->vport);
196 return ret;
197 }
198
199 bool bounds(float* px, float* py, float* pw, float* ph, bool stroking)
200 {
201 if (paints.empty()) return false;
202
203 auto x1 = FLT_MAX;
204 auto y1 = FLT_MAX;
205 auto x2 = -FLT_MAX;
206 auto y2 = -FLT_MAX;
207
208 for (auto paint : paints) {
209 auto x = FLT_MAX;
210 auto y = FLT_MAX;
211 auto w = 0.0f;
212 auto h = 0.0f;
213
214 if (!P(paint)->bounds(&x, &y, &w, &h, true, stroking)) continue;
215
216 //Merge regions
217 if (x < x1) x1 = x;
218 if (x2 < x + w) x2 = (x + w);
219 if (y < y1) y1 = y;
220 if (y2 < y + h) y2 = (y + h);
221 }
222
223 if (px) *px = x1;
224 if (py) *py = y1;
225 if (pw) *pw = (x2 - x1);
226 if (ph) *ph = (y2 - y1);
227
228 return true;
229 }
230
231 Paint* duplicate(Paint* ret)
232 {
233 if (ret) TVGERR("RENDERER", "TODO: duplicate()");
234
235 auto scene = Scene::gen().release();
236 auto dup = scene->pImpl;
237
238 for (auto paint : paints) {
239 auto cdup = paint->duplicate();
240 P(cdup)->ref();
241 dup->paints.push_back(cdup);
242 }
243
244 if (effects) TVGERR("RENDERER", "TODO: Duplicate Effects?");
245
246 return scene;
247 }
248
249 void clear(bool free)
250 {
251 for (auto paint : paints) {
252 if (P(paint)->unref() == 0 && free) delete(paint);
253 }
254 paints.clear();
255 }
256
257 Iterator* iterator()
258 {
259 return new SceneIterator(&paints);
260 }
261
262 Result resetEffects();
263};
