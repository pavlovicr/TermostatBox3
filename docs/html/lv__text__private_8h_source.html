146static inline bool lv_text_is_break_char(uint32_t letter)
147{
148 uint8_t i;
149 bool ret = false;
150
151 /*Compare the letter to TXT_BREAK_CHARS*/
152 for(i = 0; LV_TXT_BREAK_CHARS[i] != '\0'; i++) {
153 if(letter == (uint32_t)LV_TXT_BREAK_CHARS[i]) {
154 ret = true; /*If match then it is break char*/
155 break;
156 }
157 }
158
159 return ret;
160}

167static inline bool lv_text_is_a_word(uint32_t letter)
168{
169 /*Cheap check on invalid letter*/
170 if(letter == 0) return false;
171
172 /*CJK Unified Ideographs*/
173 if(letter >= 0x4E00 && letter <= 0x9FFF) {
174 return true;
175 }
176
177 /*Fullwidth ASCII variants*/
178 if(letter >= 0xFF01 && letter <= 0xFF5E) {
179 return true;
180 }
181
182 /*CJK symbols and punctuation*/
183 if(letter >= 0x3000 && letter <= 0x303F) {
184 return true;
185 }
186
187 /*CJK Radicals Supplement*/
188 if(letter >= 0x2E80 && letter <= 0x2EFF) {
189 return true;
190 }
191
192 /*CJK Strokes*/
193 if(letter >= 0x31C0 && letter <= 0x31EF) {
194 return true;
195 }
196
197 /*Hiragana and Katakana*/
198 if(letter >= 0x3040 && letter <= 0x30FF) {
199 return true;
200 }
201
202 /*Chinese Vertical Forms*/
203 if(letter >= 0xFE10 && letter <= 0xFE1F) {
204 return true;
205 }
206
207 /*CJK Compatibility Forms*/
208 if(letter >= 0xFE30 && letter <= 0xFE4F) {
209 return true;
210 }
211
212 return false;
213}

222static inline bool lv_text_is_marker(uint32_t letter)
223{
224 if(letter < 0x20) return true;
225
226 /*U+061C ARABIC LETTER MARK, see https://www.compart.com/en/unicode/block/U+0600*/
227 if(letter == 0x061C) return true;
228
229 /*U+115F HANGUL CHOSEONG FILLER, See https://www.compart.com/en/unicode/block/U+1100*/
230 if(letter == 0x115F) return true;
231 /*U+1160 HANGUL JUNGSEONG FILLER*/
232 if(letter == 0x1160) return true;
233
234 /*See https://www.compart.com/en/unicode/block/U+1800*/
235 if(letter >= 0x180B && letter <= 0x180E) return true;
236
237 /*See https://www.compart.com/en/unicode/block/U+2000*/
238 if(letter >= 0x200B && letter <= 0x200F) return true;
239 if(letter >= 0x2028 && letter <= 0x202F) return true;
240 if(letter >= 0x205F && letter <= 0x206F) return true;
241
242 /*U+FEFF ZERO WIDTH NO-BREAK SPACE, see https://www.compart.com/en/unicode/block/U+FE70*/
243 if(letter == 0xFEFF) return true;
244
245 if(letter == 0xF8FF) return true; /*LV_SYMBOL_DUMMY*/
246
247 return false;
248}
