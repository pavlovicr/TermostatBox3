29FASTGLTF_EXPORT inline auto getFastLocalTransformMatrix(const Node & node)
30{
31 return visit_exhaustive(visitor {
32 [&](const math::fmat4x4 & matrix)
33 {
34 return matrix;
35 },
36 [&](const TRS & trs)
37 {
38 math::fmat4x4 matrix = math::fmat4x4();
39 float sx = trs.scale[0], sy = trs.scale[1], sz = trs.scale[2];
40 float qx = trs.rotation[0], qy = trs.rotation[1], qz = trs.rotation[2], qw = trs.rotation[3];
41 float x2 = qx + qx, y2 = qy + qy, z2 = qz + qz;
42 float xx = qx * x2, xy = qx * y2, xz = qx * z2;
43 float yy = qy * y2, yz = qy * z2, zz = qz * z2;
44 float wx = qw * x2, wy = qw * y2, wz = qw * z2;
45 matrix[0][0] = (1 - (yy + zz)) * sx;
46 matrix[0][1] = (xy + wz) * sx;
47 matrix[0][2] = (xz - wy) * sx;
48 matrix[1][0] = (xy - wz) * sy;
49 matrix[1][1] = (1 - (xx + zz)) * sy;
50 matrix[1][2] = (yz + wx) * sy;
51 matrix[2][0] = (xz + wy) * sz;
52 matrix[2][1] = (yz - wx) * sz;
53 matrix[2][2] = (1 - (xx + yy)) * sz;
54 matrix[3][0] = trs.translation[0];
55 matrix[3][1] = trs.translation[1];
56 matrix[3][2] = trs.translation[2];
57 matrix[0][3] = 0.f;
58 matrix[1][3] = 0.f;
59 matrix[2][3] = 0.f;
60 matrix[3][3] = 1.f;
61 return matrix;
62 }
63 }, node.transform);
64}

70FASTGLTF_EXPORT inline void removeScale(fastgltf::math::fmat4x4 & matrix)
71{
72 auto scale = math::fvec3(length(matrix.col(0)), length(matrix.col(1)), length(matrix.col(2)));
73 if((fabs(scale.x()) > 0.00001f) && (fabs(scale.y()) > 0.00001f) && (fabs(scale.z()) > 0.00001f)) {
74 matrix.col(0) /= scale.x();
75 matrix.col(1) /= scale.y();
76 matrix.col(2) /= scale.z();
77 }

86void namegen_iterate_scene_nodes(AssetType &&asset,
87 std::size_t sceneIndex,
88 Callback callback)
89{
90 auto & scene = asset.scenes[sceneIndex];
91
92 std::string _id = std::string("");
93 std::string _ip = std::string("");
94 if(asset.scenes.size() > 1) {
95 _id = "scene_" + std::to_string(sceneIndex);
96 _ip = std::to_string(sceneIndex);
97 }
98 auto function = [&](std::size_t nodeIndex, std::string & parentId,
99 std::string & parentIp, std::size_t __child_index,
100 auto & self) -> void {
101 assert(asset.nodes.size() > nodeIndex);
102 auto & node = asset.nodes[nodeIndex];
103 std::string _nodeId =
104 parentId + std::string("/") + std::string(node.name);
105 std::string _nodeIp = parentIp + std::string(".") +
106 std::to_string(__child_index);
107 std::invoke(callback, node, _nodeId, _nodeIp, nodeIndex,
108 __child_index);
109 std::size_t ____child_index = 0;
110 for(auto & child : node.children) {
111 self(child, _nodeId, _nodeIp, ____child_index, self);
112 ____child_index += 1;
113 }
114 };
115 std::size_t child_index = 0;
116 for(auto & sceneNode : scene.nodeIndices) {
117 function(sceneNode, _id, _ip, child_index, function);
118 child_index += 1;
119 }
120}

126void
127findlight_iterate_scene_nodes(AssetType &&asset, std::size_t sceneIndex,
128 math::fmat4x4 * initial, Callback callback)
129{
130 auto & scene = asset.scenes[sceneIndex];
131 auto function = [&](std::size_t nodeIndex,
132 math::fmat4x4 & parentWorldMatrix,
133 auto & self) -> void {
134 assert(asset.nodes.size() > nodeIndex);
135 auto & node = asset.nodes[nodeIndex];
136 auto _localMat = getFastLocalTransformMatrix(node);
137 std::invoke(callback, node, parentWorldMatrix, _localMat);
138 for(auto & child : node.children) {
139 math::fmat4x4 _parentWorldTemp =
140 parentWorldMatrix * _localMat;
141 self(child, _parentWorldTemp, self);
142 }
143 };
144 for(auto & sceneNode : scene.nodeIndices) {
145 auto tmat2 = fastgltf::math::fmat4x4(*initial);
146 function(sceneNode, tmat2, function);
147 }

150inline void custom_iterate_scene_nodes(lv_gltf_model_t * model, std::size_t sceneIndex, math::fmat4x4 * initial,
151 Callback callback)
152{
153 auto & scene = model->asset.scenes[sceneIndex];
154
155 auto invoke_cb = [&](std::size_t node_index, math::fmat4x4 & parent_world_matrix,
156 auto & self) -> void {
157 lv_gltf_model_node_t * node = (lv_gltf_model_node_t *)lv_array_at(&model->nodes, node_index);
158 fastgltf::Node & fastgltf_node = *node->fastgltf_node;
159 LV_ASSERT(node->fastgltf_node == &fastgltf_node);
160
161 auto local_matrix = getFastLocalTransformMatrix(fastgltf_node);
162 std::invoke(callback, node, parent_world_matrix, local_matrix);
163
164 uint32_t num_children = fastgltf_node.children.size();
165 if(num_children == 0) {
166 return;
167 }
168 math::fmat4x4 new_parent_world_matrix = parent_world_matrix * local_matrix;
169 if(num_children == 1)
170 {
171 self(fastgltf_node.children[0], new_parent_world_matrix, self);
172 return;
173 }
174 math::fmat4x4 parent_world_matrix_tmp_copy = math::fmat4x4(new_parent_world_matrix);
175 for(auto & child : fastgltf_node.children)
176 {
177 self(child, parent_world_matrix_tmp_copy, self);
178 }
179 };
180
181 for(size_t i = 0 ; i < scene.nodeIndices.size(); ++i) {
182 invoke_cb(scene.nodeIndices[i], *initial, invoke_cb);
183 }
184}
