31/*
32 * Macros used to set cubic-bezier anim parameter.
33 * Parameters come from https://easings.net/
34 *
35 * Usage:
36 *
37 * lv_anim_t a;
38 * lv_anim_init(&a);
39 * ...
40 * lv_anim_set_path_cb(&a, lv_anim_path_custom_bezier3);
41 * LV_ANIM_SET_EASE_IN_SINE(&a); //Set cubic-bezier anim parameter to easeInSine
42 * ...
43 * lv_anim_start(&a);
44 */
45
46#define _PARA(a, x1, y1, x2, y2) ((a)->parameter.bezier3 = \
47(lv_anim_bezier3_para_t) { \
48 LV_BEZIER_VAL_FLOAT(x1), LV_BEZIER_VAL_FLOAT(y1), \
49 LV_BEZIER_VAL_FLOAT(x2), LV_BEZIER_VAL_FLOAT(y2) } \
50 )
