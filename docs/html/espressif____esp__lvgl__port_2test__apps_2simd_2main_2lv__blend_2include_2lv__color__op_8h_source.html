48static inline lv_color_t LV_ATTRIBUTE_FAST_MEM lv_color_mix(lv_color_t c1, lv_color_t c2, uint8_t mix)
49{
50 lv_color_t ret;
51
52 ret.red = LV_UDIV255((uint16_t)c1.red * mix + c2.red * (255 - mix) + LV_COLOR_MIX_ROUND_OFS);
53 ret.green = LV_UDIV255((uint16_t)c1.green * mix + c2.green * (255 - mix) + LV_COLOR_MIX_ROUND_OFS);
54 ret.blue = LV_UDIV255((uint16_t)c1.blue * mix + c2.blue * (255 - mix) + LV_COLOR_MIX_ROUND_OFS);
55 return ret;
56}

66static inline lv_color32_t lv_color_mix32(lv_color32_t fg, lv_color32_t bg)
67{
68 if (fg.alpha >= LV_OPA_MAX) {
69 fg.alpha = bg.alpha;
70 return fg;
71 }
72 if (fg.alpha <= LV_OPA_MIN) {
73 return bg;
