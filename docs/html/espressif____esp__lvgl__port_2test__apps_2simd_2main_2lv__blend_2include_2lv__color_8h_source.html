225static inline uint16_t LV_ATTRIBUTE_FAST_MEM lv_color_16_16_mix(uint16_t c1, uint16_t c2, uint8_t mix)
226{
227 if (mix == 255) {
228 return c1;
229 }
230 if (mix == 0) {
231 return c2;
232 }
233 if (c1 == c2) {
234 return c1;
235 }
236
237 uint16_t ret;
238
239 /* Source: https://stackoverflow.com/a/50012418/1999969*/
240 mix = (uint32_t)((uint32_t)mix + 4) >> 3;
241
242 /*0x7E0F81F = 0b00000111111000001111100000011111*/
243 uint32_t bg = (uint32_t)(c2 | ((uint32_t)c2 << 16)) & 0x7E0F81F;
244 uint32_t fg = (uint32_t)(c1 | ((uint32_t)c1 << 16)) & 0x7E0F81F;
245 uint32_t result = ((((fg - bg) * mix) >> 5) + bg) & 0x7E0F81F;
246 ret = (uint16_t)(result >> 16) | result;
247
248 return ret;
249}
