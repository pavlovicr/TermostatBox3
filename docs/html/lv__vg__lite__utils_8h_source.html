47 do { \
48 if(lv_vg_lite_is_dump_param_enabled()) { \
49 LV_LOG_USER("Call '" #expr "', Parameter:"); \
50 dump_param; \
51 } \
52 vg_lite_error_t error = expr; \

54 LV_LOG_ERROR("Execute '" #expr "' error: %d", (int)error); \
55 lv_vg_lite_error_dump_info(error); \
56 dump_param; \
57 LV_VG_LITE_ASSERT(false); \
58 } \
59 } while (0)
