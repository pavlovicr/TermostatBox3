38template <typename T>
39[[nodiscard]] fastgltf::math::quat<T> lv_gltf_math_euler_to_quaternion(T P, T Y, T R)
40{
41 // Convert degrees to radians if necessary
42 // roll = roll * (M_PI / 180.0);
43 // pitch = pitch * (M_PI / 180.0);
44 // yaw = yaw * (M_PI / 180.0);
45 T H = T(0.5);
46 Y *= H;
47 P *= H;
48 R *= H;
49 T cy = cos(Y), sy = sin(Y), cp = cos(P), sp = sin(P), cr = cos(R), sr = sin(R);
50 T cr_cp = cr * cp, sp_sy = sp * sy, sr_cp = sr * cp, sp_cy = sp * cy;
51 return fastgltf::math::quat<T>(
52 sr_cp * cy - cr * sp_sy, // X
53 sr_cp * sy + cr * sp_cy, // Y
54 cr_cp * sy - sr * sp_cy, // Z
55 cr_cp * cy + sr * sp_sy // W
56 );
57}

59template <typename T>
60[[nodiscard]] fastgltf::math::vec<T, 3> lv_gltf_math_quaternion_to_euler(fastgltf::math::quat<T> q)
61{
62 T Q11 = q[1] * q[1];
63 // Roll (Z)
64 T sinr_cosp = T(2.0) * (q[3] * q[0] + q[1] * q[2]);
65 T cosr_cosp = T(1.0) - T(2.0) * (q[0] * q[0] + Q11);
66 // Pitch (X)
67 T sinp = T(2.0) * (q[3] * q[1] - q[2] * q[0]);
68 // Yaw (Y)
69 T siny_cosp = T(2.0) * (q[3] * q[2] + q[0] * q[1]);
70 T cosy_cosp = T(1.0) - T(2.0) * (Q11 + q[2] * q[2]);
71
72 return fastgltf::math::vec<T, 3>(
73 (std::abs(sinp) >= T(1)) ? std::copysign(T(M_PI) / T(2), sinp) : std::asin(sinp),
74 std::atan2(siny_cosp, cosy_cosp),
75 std::atan2(sinr_cosp, cosr_cosp)
76 );
77}
