62/* This structure describes one color plane. Some YUV formats may support
63 * up to 4 planes (although they probably wouldn't be used on LCD hardware).
64 * The framebuffer driver provides the video memory address in its
65 * corresponding fb_planeinfo_s structure. The LCD driver, instead, provides
66 * methods to transfer data to/from the LCD color plane.
67 */
68
69struct lcd_planeinfo_s {
70 /* This is working memory allocated by the LCD driver for each LCD device
71 * and for each color plane. This memory will hold one raster line of
72 * data. The size of the allocated run buffer must therefore be at least
73 * (bpp * xres / 8). Actual alignment of the buffer must conform to the
74 * bitwidth of the underlying pixel type.
75 *
76 * If there are multiple planes, they may share the same working buffer
77 * because different planes will not be operate on concurrently. However,
78 * if there are multiple LCD devices, they must each have unique run
79 * buffers.
80 */
81
82 uint8_t * buffer;
83
84 /* This is the number of bits in one pixel. This may be one of {1, 2, 4,
85 * 8, 16, 24, or 32} unless support for one or more of those resolutions
86 * has been disabled.
87 */

89 uint8_t bpp;
