170/*
171 * PGA1 gain registers REG0x72 (ADCL / MIC1 path) and REG0x73 (ADCR / MIC2 path):
172 * bits [6:4] are the 3-bit analog input mux. Datasheet binary vs software decimal:
173 * 0b001 (1) – differential (ADCL: MIC1P–MIC1N; ADCR: MIC2N–MIC1P; see enumerators)
174 * 0b101 (5) – single-ended (ADCL: MIC1P; ADCR: MIC2N)
175 * 0b110 (6) – single-ended (ADCL: MIC2P; ADCR: MIC1N)
176 * Bits [3:0] are PGA gain steps (es8389_mic_gain_t); bit7 is reverse. Same code selects
177 * different pads on L vs R; see each enumerator.
178 */
179typedef enum {
180 /* ADCL */
181 ES8389_ADC_INSEL_L_DIFF_MIC1P1N = 1, /* Diff MIC1P-MIC1N (ES8390: LIN1-RIN1) */
182 ES8389_ADC_INSEL_L_SINGLE_MIC1P = 5, /* Single-ended MIC1P (ES8390: LIN1) */
183 ES8389_ADC_INSEL_L_SINGLE_MIC2P = 6, /* Single-ended MIC2P (ES8390: LIN2) */
184 /* ADCR */
185 ES8389_ADC_INSEL_R_DIFF_MIC2N2P = 1, /* Diff MIC2N-MIC1P (ES8390: RIN2-LIN2) */
186 ES8389_ADC_INSEL_R_SINGLE_MIC2N = 5, /* Single-ended MIC2N (ES8390: RIN2) */
187 ES8389_ADC_INSEL_R_SINGLE_MIC1N = 6, /* Single-ended MIC1N (ES8390: RIN1) */
188} es8389_adc_insel_t;
189
190#define ES8389_ADCL_InputSel ES8389_ADC_INSEL_L_DIFF_MIC1P1N
191#define ES8389_ADCR_InputSel ES8389_ADC_INSEL_R_DIFF_MIC2N2P
