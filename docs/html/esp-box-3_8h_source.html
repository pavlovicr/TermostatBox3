140/* PMOD */
141/*
142 * PMOD interface (peripheral module interface) is an open standard defined by Digilent Inc.
143 * for peripherals used with FPGA or microcontroller development boards.
144 *
145 * ESP-BOX contains two double PMOD connectors, protected with ESD protection diodes.
146 * Power pins are on 3.3V.
147 *
148 * Double PMOD Connectors on ESP-BOX-3 dock are labeled as follows:
149 * |------------|
150 * | IO1 IO5 |
151 * | IO2 IO6 |
152 * | IO3 IO7 |
153 * | IO4 IO8 |
154 * |------------|
155 * | GND GND |
156 * | 3V3 3V3 |
157 * |------------|
158 */

163#define BSP_PMOD1_IO5 GPIO_NUM_21
164#define BSP_PMOD1_IO6 BSP_USB_NEG
165#define BSP_PMOD1_IO7 GPIO_NUM_38

198/**************************************************************************************************
199 *
200 * I2S audio interface
201 *
202 * There are two devices connected to the I2S peripheral:
203 * - Codec ES8311 for output (playback) path
204 * - ADC ES7210 for input (recording) path
205 *
206 * For speaker initialization use bsp_audio_codec_speaker_init() which is inside initialize I2S with bsp_audio_init().
207 * For microphone initialization use bsp_audio_codec_microphone_init() which is inside initialize I2S with bsp_audio_init().
208 * After speaker or microphone initialization, use functions from esp_codec_dev for play/record audio.
209 * Example audio play:
210 * \code{.c}
211 * esp_codec_dev_set_out_vol(spk_codec_dev, DEFAULT_VOLUME);
212 * esp_codec_dev_open(spk_codec_dev, &fs);
213 * esp_codec_dev_write(spk_codec_dev, wav_bytes, bytes_read_from_spiffs);
214 * esp_codec_dev_close(spk_codec_dev);
215 * \endcode
216 **************************************************************************************************/
