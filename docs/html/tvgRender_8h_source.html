106struct RenderStroke
107{
108 float width = 0.0f;
109 uint8_t color[4] = {0, 0, 0, 0};
110 Fill *fill = nullptr;
111 float* dashPattern = nullptr;
112 uint32_t dashCnt = 0;
113 float dashOffset = 0.0f;
114 StrokeCap cap = StrokeCap::Square;
115 StrokeJoin join = StrokeJoin::Bevel;
116 float miterlimit = 4.0f;
117 bool strokeFirst = false;
118
119 struct {
120 float begin = 0.0f;
121 float end = 1.0f;
122 bool simultaneous = true;
123 } trim;
124
125 void operator=(const RenderStroke& rhs)
126 {
127 width = rhs.width;
128
129 memcpy(color, rhs.color, sizeof(color));
130
131 delete(fill);
132 if (rhs.fill) fill = rhs.fill->duplicate();
133 else fill = nullptr;
134
135 lv_free(dashPattern);
136 if (rhs.dashCnt > 0) {
137 dashPattern = static_cast<float*>(lv_malloc(sizeof(float) * rhs.dashCnt));
138 LV_ASSERT_MALLOC(dashPattern);
139 memcpy(dashPattern, rhs.dashPattern, sizeof(float) * rhs.dashCnt);
140 } else {
141 dashPattern = nullptr;
142 }
143 dashCnt = rhs.dashCnt;
144 miterlimit = rhs.miterlimit;
145 cap = rhs.cap;
146 join = rhs.join;
147 strokeFirst = rhs.strokeFirst;
148 trim = rhs.trim;
149 }
150
151 bool strokeTrim(float& begin, float& end) const
152 {
153 begin = trim.begin;
154 end = trim.end;
155
156 if (fabsf(end - begin) >= 1.0f) {
157 begin = 0.0f;
158 end = 1.0f;
159 return false;
160 }
161
162 auto loop = true;
163
164 if (begin > 1.0f && end > 1.0f) loop = false;
165 if (begin < 0.0f && end < 0.0f) loop = false;
166 if (begin >= 0.0f && begin <= 1.0f && end >= 0.0f && end <= 1.0f) loop = false;
167
168 if (begin > 1.0f) begin -= 1.0f;
169 if (begin < 0.0f) begin += 1.0f;
170 if (end > 1.0f) end -= 1.0f;
171 if (end < 0.0f) end += 1.0f;
172
173 if ((loop && begin < end) || (!loop && begin > end)) std::swap(begin, end);
174 return true;
175 }
176
177 ~RenderStroke()
178 {
179 lv_free(dashPattern);
180 delete(fill);
181 }
182};

184struct RenderShape

204 {
205 if (r) *r = color[0];

207 if (b) *b = color[2];

211 float strokeWidth() const
212 {
213 if (!stroke) return 0;
214 return stroke->width;
215 }
216
217 bool strokeTrim() const
218 {
219 if (!stroke) return false;
220 if (stroke->trim.begin == 0.0f && stroke->trim.end == 1.0f) return false;
221 if (fabsf(stroke->trim.end - stroke->trim.begin) >= 1.0f) return false;
222 return true;
223 }
224
225 bool strokeColor(uint8_t* r, uint8_t* g, uint8_t* b, uint8_t* a) const
226 {
227 if (!stroke) return false;
228
229 if (r) *r = stroke->color[0];
230 if (g) *g = stroke->color[1];
231 if (b) *b = stroke->color[2];
232 if (a) *a = stroke->color[3];
233
234 return true;
235 }
236
237 const Fill* strokeFill() const
238 {
239 if (!stroke) return nullptr;
240 return stroke->fill;
241 }
242
243 uint32_t strokeDash(const float** dashPattern, float* offset) const
244 {
245 if (!stroke) return 0;
246 if (dashPattern) *dashPattern = stroke->dashPattern;
247 if (offset) *offset = stroke->dashOffset;
248 return stroke->dashCnt;
249 }
250
251 StrokeCap strokeCap() const
252 {
253 if (!stroke) return StrokeCap::Square;
254 return stroke->cap;
255 }
256
257 StrokeJoin strokeJoin() const
258 {
259 if (!stroke) return StrokeJoin::Bevel;
260 return stroke->join;
261 }
262
263 float strokeMiterlimit() const

283struct RenderEffectGaussian : RenderEffect
284{
285 float sigma;
286 uint8_t direction; //0: both, 1: horizontal, 2: vertical
287 uint8_t border; //0: duplicate, 1: wrap
288 uint8_t quality; //0 ~ 100 (optional)
289
290 static RenderEffectGaussian* gen(va_list& args)
291 {
292 auto sigma = (float) va_arg(args, double);
293 if (sigma <= 0) return nullptr;
294
295 auto inst = new RenderEffectGaussian;
296 inst->sigma = sigma;
297 inst->direction = std::min(va_arg(args, int), 2);
298 inst->border = std::min(va_arg(args, int), 1);
299 inst->quality = std::min(va_arg(args, int), 100);
300 inst->type = SceneEffect::GaussianBlur;
301 return inst;
302 }

341static inline bool MASK_REGION_MERGING(CompositeMethod method)
342{
343 switch(method) {
344 case CompositeMethod::AlphaMask:
345 case CompositeMethod::InvAlphaMask:
346 case CompositeMethod::LumaMask:
347 case CompositeMethod::InvLumaMask:
348 case CompositeMethod::SubtractMask:
349 case CompositeMethod::IntersectMask:
350 return false;
351 //these might expand the rendering region
352 case CompositeMethod::AddMask:
353 case CompositeMethod::DifferenceMask:
354 case CompositeMethod::LightenMask:
355 case CompositeMethod::DarkenMask:
356 return true;
357 default:
358 TVGERR("RENDERER", "Unsupported Composite Method! = %d", (int)method);
359 return false;
360 }
361}

363static inline uint8_t CHANNEL_SIZE(ColorSpace cs)
364{
365 switch(cs) {
366 case ColorSpace::ABGR8888:
367 case ColorSpace::ABGR8888S:
368 case ColorSpace::ARGB8888:
369 case ColorSpace::ARGB8888S:
370 return sizeof(uint32_t);
371 case ColorSpace::Grayscale8:
372 return sizeof(uint8_t);
373 case ColorSpace::Unsupported:
374 default:
375 TVGERR("RENDERER", "Unsupported Channel Size! = %d", (int)cs);
376 return 0;
377 }

381{
382 switch(method) {

384 case CompositeMethod::InvAlphaMask:
385 case CompositeMethod::AddMask:
386 case CompositeMethod::DifferenceMask:
387 case CompositeMethod::SubtractMask:

397 TVGERR("RENDERER", "Unsupported Composite Size! = %d", (int)method);
398 return ColorSpace::Unsupported;
399 }
