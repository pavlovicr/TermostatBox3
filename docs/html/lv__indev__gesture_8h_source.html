49/* The states of a gesture recognizer */
50typedef enum {
51 LV_INDEV_GESTURE_STATE_NONE = 0, /* Beginning & end */
52 LV_INDEV_GESTURE_STATE_ONGOING, /* Set when there is a probability */
53 LV_INDEV_GESTURE_STATE_RECOGNIZED, /* Recognized, the event will contain touch info */
54 LV_INDEV_GESTURE_STATE_ENDED, /* A recognized gesture has ended */
55 LV_INDEV_GESTURE_STATE_CANCELED, /* Canceled - usually a finger is lifted */
56} lv_indev_gesture_state_t;
57
58/* Data structures for touch events - used to repsensent a libinput event */
59/* Emitted by devices capable of tracking identifiable contacts (type B) */
60struct lv_indev_touch_data {
61 lv_point_t point; /* Coordinates of the touch */
62 lv_indev_state_t state; /* The state i.e PRESSED or RELEASED */
63 uint8_t id; /* Identification/slot of the contact point */
64 uint32_t timestamp; /* Timestamp in milliseconds */
65};

67/* Gesture recognizer */
68struct lv_indev_gesture_recognizer {
69 lv_indev_gesture_type_t type; /* The detected gesture type */
70 lv_indev_gesture_state_t state; /* The gesture state ongoing, recognized */
71 lv_indev_gesture_t * info; /* Information on the motion of each touch point */
72 float scale; /* Relevant for the pinch gesture */
73 float rotation; /* Relevant for rotation */
74 float distance; /* Relevant for swipes */
75 float speed;
76 lv_dir_t two_fingers_swipe_dir; /* Relevant for swipes */
77
78 lv_indev_gesture_configuration_t * config; /* The recognizer config, containing the gestures
79 thresholds */
80 lv_recognizer_func_t recog_fn; /* The recognizer function that this recongnizer must execute */
81};
