152 /*The properties are listed in a special order to make caching more effective.
153 *Groups are used when LV_OBJ_STYLE_CACHE is enabled.
154 *If a property from a groups is set, a bit will be set in
155 *obj->style_main_prop_is_set and style_other_prop_is_set
156 *to indicate that the property is set.
157 *
158 *Strategy 1: group related properties. E.g. if no "border" properties are set
159 * they will be skipped quickly.
160 *
161 *Strategy 2: group common property with rarely used properties. This way
162 * the common property is cached properly and it's rarely affected
163 * by other props. The other props are cached in an sub-optimal way,

529static inline lv_style_res_t lv_style_get_prop_inlined(const lv_style_t * style, lv_style_prop_t prop,
530 lv_style_value_t * value)
531{
532 if(lv_style_is_const(style)) {
533 lv_style_const_prop_t * props = (lv_style_const_prop_t *)style->values_and_props;
534 uint32_t i;
535 for(i = 0; props[i].prop != LV_STYLE_PROP_INV; i++) {
536 if(props[i].prop == prop) {
537 *value = props[i].value;
538 return LV_STYLE_RES_FOUND;
539 }
540 }
541 }
542 else {
543 lv_style_prop_t * props = (lv_style_prop_t *)style->values_and_props + style->prop_cnt * sizeof(lv_style_value_t);
544 uint32_t i;
545 for(i = 0; i < style->prop_cnt; i++) {
546 if(props[i] == prop) {
547 lv_style_value_t * values = (lv_style_value_t *)style->values_and_props;
548 *value = values[i];
549 return LV_STYLE_RES_FOUND;
550 }
551 }
552 }
553 return LV_STYLE_RES_NOT_FOUND;
554}
