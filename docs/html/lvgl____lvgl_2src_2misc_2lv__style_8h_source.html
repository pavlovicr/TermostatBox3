489static inline lv_style_res_t lv_style_get_prop_inlined(const lv_style_t * style, lv_style_prop_t prop,
490 lv_style_value_t * value)
491{
492 if(lv_style_is_const(style)) {
493 lv_style_const_prop_t * props = (lv_style_const_prop_t *)style->values_and_props;
494 uint32_t i;
495 for(i = 0; props[i].prop != LV_STYLE_PROP_INV; i++) {
496 if(props[i].prop == prop) {
497 *value = props[i].value;
498 return LV_STYLE_RES_FOUND;
499 }
500 }
501 }
502 else {
503 lv_style_prop_t * props = (lv_style_prop_t *)style->values_and_props + style->prop_cnt * sizeof(lv_style_value_t);
504 uint32_t i;
505 for(i = 0; i < style->prop_cnt; i++) {
506 if(props[i] == prop) {
507 lv_style_value_t * values = (lv_style_value_t *)style->values_and_props;
508 *value = values[i];
509 return LV_STYLE_RES_FOUND;
510 }
511 }
512 }
513 return LV_STYLE_RES_NOT_FOUND;
514}
