\hypertarget{md_managed__components_2lvgl____lvgl_2scripts_2gdb_2lvglgdb_2lvgl_2misc_2README_autotoc_md584}{}\doxysection{\texorpdfstring{Type Conversion Support}{Type Conversion Support}}\label{md_managed__components_2lvgl____lvgl_2scripts_2gdb_2lvglgdb_2lvgl_2misc_2README_autotoc_md584}
All wrapper classes support multiple input types\+:


\begin{DoxyItemize}
\item {\ttfamily gdb.\+Value} objects (from GDB expressions)
\item {\ttfamily Value} objects (from lvglgdb.\+value module)
\item Raw pointers as integers (e.\+g., {\ttfamily 0x12345678})
\end{DoxyItemize}

This allows flexible usage patterns\+:


\begin{DoxyCode}{0}
\DoxyCodeLine{\#\ From\ GDB\ variable}
\DoxyCodeLine{entry\ =\ LVCacheEntry(my\_entry)}
\DoxyCodeLine{}
\DoxyCodeLine{\#\ From\ raw\ memory\ address}
\DoxyCodeLine{entry\ =\ LVCacheEntry(0x7fff12345678)}
\DoxyCodeLine{}
\DoxyCodeLine{\#\ From\ GDB\ expression\ result}
\DoxyCodeLine{entry\ =\ LVCacheEntry(gdb.parse\_and\_eval("{}*(lv\_cache\_entry\_t*)0x7fff12345678"{}))}

\end{DoxyCode}
